1. 安装Node.js和npm
bashsudo apt update
sudo apt install nodejs npm

2. 创建项目目录
bashmkdir chat-app
cd chat-app

3. 创建文件结构
# 创建public目录
mkdir public

# 创建package.json
nano package.json

# 创建服务器文件
nano server.js

# 创建前端页面
nano public/index.html

4. 安装依赖
npm install
npm install multer

5. 启动服务器
node server.js

6. 访问应用
http://你的服务器IP:3000

7. 开机启动
-创建服务文件
vi /etc/systemd/system/chat-app.service
2. 添加以下内容：
将 WorkingDirectory=xxxxxxxx替换为你的实际路径
##########################
[Unit]
Description=Simple Chat App
After=network.target

[Service]
Type=simple
User=root
WorkingDirectory=/root/chat-app
ExecStart=/usr/bin/node server.js
Restart=always
RestartSec=10
Environment=NODE_ENV=production
Environment=PORT=3000

[Install]
WantedBy=multi-user.target
###########################

# 启用开机启动
sudo systemctl enable chat-app.service
